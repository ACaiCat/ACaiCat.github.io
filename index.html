GitHub Actions
#

name: Grow cotton

on:
  push:
    branches: [ "main" ]

permissions:
  contents: write

jobs:
  grow:
    runs-on: ubuntu-latest

    strategy:
        matrix:
            ai: ["deepseek.v3", "gpt-4", "claude-4.5"]

    steps:
    - name: Grow cotton
      uses: ACaiCat/cotton@v114514
      with:
        worker: ${{ matrix.ai }}
        go_work: true
        output_path: output/

    - name: Upload cottons
      uses: actions/upload-artifact@v4
      with:
        name: cotton-${{ matrix.ai }}
        path: output/
